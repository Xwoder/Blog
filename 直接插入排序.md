# 直接插入排序

## Java实现

```Java
import java.util.Arrays;

public class Main {

    public static void main(String[] args) {
        Integer[] array = new Integer[]{104, 363, 463, 936, 514, 608, 511, 408, 993, 220};

        // 输出
        // [104, 363, 463, 936, 514, 608, 511, 408, 993, 220]
        System.out.println(Arrays.toString(array));

        directInsertionSort(array);

        // 输出
        // [104, 220, 363, 408, 463, 511, 514, 608, 936, 993]
        System.out.println(Arrays.toString(array));
    }

    /**
     * 直接插入排序
     */
    private static void directInsertionSort(Integer[] array) {
        for (int i = 1; i < array.length; i++) {
            Integer target = array[i];

            int j;
            for (j = i; j >= 1 && array[j - 1] > target; j--) {
                array[j] = array[j - 1];
            }

            array[j] = target;
        }
    }

    private static void swap(Integer[] array, Integer i, Integer j) {
        var temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }
}
```
