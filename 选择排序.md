## Java实现

```Java
import java.util.Arrays;

public class Main {

    public static void main(String[] args) {
        Integer[] array = new Integer[]{961, 587, 897, 934, 159, 470, 793, 380, 178, 382};

        // 输出
        // [961, 587, 897, 934, 159, 470, 793, 380, 178, 382]
        System.out.println(Arrays.toString(array));

        selectionSort(array);

        // 输出
        // [159, 178, 380, 382, 470, 587, 793, 897, 934, 961]
        System.out.println(Arrays.toString(array));
    }

    /**
     * 选择排序
     */
    private static void selectionSort(Integer[] array) {
        for (int i = 1; i <= array.length - 1; i++) {
            int targetIndex = 0;
            for (int j = 1; j <= array.length - i; j++) {
                if (array[j] > array[targetIndex]) {
                    targetIndex = j;
                }
            }

            if (targetIndex != array.length - i) {
                swap(array, targetIndex, array.length - i);
            }
        }
    }

    private static void swap(Integer[] array, Integer i, Integer j) {
        var temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }
}
```
