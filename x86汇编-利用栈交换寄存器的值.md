```asm
assume cs:codeseg ss:stackseg

stackseg segment
    dw 2 dup(0)
stackseg ends

codeseg segment

start:
    ; 设置栈段
    mov ax, stackseg
    mov ss, ax
    mov sp, 4

    ; 初始化寄存器
    mov ax, 1122h
    mov bx, 3344h

    ; 交换寄存器的值
    push ax
    push bx
    pop ax
    pop bx

    ; 退出
    mov ah, 4ch
    int 21h

codeseg ends

end start
```
